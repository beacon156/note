查询根据时间累计数据:
	SELECT aa.日期, aa.数值, SUM(lj.数值) AS 累计 
	FROM tb lj INNER JOIN
	(SELECT *    FROM tb) aa ON lj.日期 <= aa.日期
	WHERE (aa.日期 <= '2005-05-04')   
	GROUP BY aa.日期, aa.数值
	ORDER BY aa.日期
